package com.f1hiberusmartin.estrategia;
import lombok.Getter;
import lombok.Setter;
import java.util.Objects;
public class Combustible {
    @Getter @Setter private String tipo;
    @Getter @Setter private float capacidad;
    public Combustible(String tipo, float capacidad) {
        if (tipo != null && !("Gasolina".equals(tipo) || "Diésel".equals(tipo))) {
            throw new IllegalArgumentException("El tipo de combustible debe ser 'Gasolina' o 'Diésel'");
        } if (capacidad < 0.0f) {
            throw new IllegalArgumentException("La capacidad no puede ser negativa");
        } this.tipo = tipo;
        this.capacidad = capacidad;
    }
    @Override public boolean equals(Object obj) {
        if (this == obj) return true;
        if (obj == null || getClass() != obj.getClass()) return false;
            Combustible other = (Combustible) obj;
            return Objects.equals(tipo, other.tipo) && Float.compare(other.capacidad, capacidad) == 0;

    }
    @Override
    public int hashCode() {
        return Objects.hash(tipo, capacidad);
    }
}
